-- DROP DATABASE SECONDAUTO;
CREATE DATABASE SECONDAUTO;

USE SECONDAUTO;

-- Definizione della tabella REGIONE
CREATE TABLE REGIONE (
    ID_regione INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
    Nome_Regione VARCHAR(100) NOT NULL
);
    
-- Definizione della tabella CITTA'
CREATE TABLE CITTA (
    ID_citta INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
    Nome_citta VARCHAR(100) NOT NULL,
    ID_regione INT,
    FOREIGN KEY (ID_regione)
        REFERENCES REGIONE (ID_regione)
);

-- Definizione della tabella CONCESSIONARIA
CREATE TABLE CONCESSIONARIA (
    ID_concessionaria INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
    ID_citta INT NOT NULL,
    Indirizzo VARCHAR(100) NOT NULL,
    Telefono VARCHAR(20) UNIQUE NOT NULL,
    Responsabile VARCHAR(100),
    FOREIGN KEY (ID_citta)
        REFERENCES CITTA (ID_citta)
);

-- Definizione della tabella AUTO
CREATE TABLE AUTO (
    Targa CHAR(07) PRIMARY KEY,
    Marca VARCHAR(50),
    Colore VARCHAR(50),
    Cilindrata INT,
    Tipo_Cambio VARCHAR(50),
    Prezzo DECIMAL(10 , 2 )
);

-- Definizione della tabella PROPRIETARI
CREATE TABLE PROPRIETARI (
    ID_proprietari INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
    Nominativo VARCHAR(100),
    Targa VARCHAR(20),
    Residenza INT,
    FOREIGN KEY (Targa)
        REFERENCES AUTO (Targa),
    FOREIGN KEY (Residenza)
        REFERENCES CITTA (ID_citta)
);

-- Definizione della tabella TRANSAZIONI
CREATE TABLE TRANSAZIONI (
    ID_transazione INT AUTO_INCREMENT PRIMARY KEY NOT NULL,
    ID_concessionaria INT,
    Targa CHAR(07),
    Data DATE NOT NULL,
    tipo_transazione ENUM('acquisto', 'vendita'),
    FOREIGN KEY (ID_concessionaria)
        REFERENCES CONCESSIONARIA (ID_concessionaria),
    FOREIGN KEY (Targa)
        REFERENCES AUTO (Targa)
);

CREATE VIEW Fatturato AS
SELECT distinct
    c.ID_concessionaria,
    SUM(CASE WHEN t.tipo_transazione = 'acquisto' THEN a.Prezzo ELSE 0 END) AS 'Fatturato Acquisti',
    SUM(CASE WHEN t.tipo_transazione = 'vendita' THEN a.Valore_vendita ELSE 0 END) AS 'Fatturato Vendite'
FROM TRANSAZIONI t
JOIN CONCESSIONARIA c ON t.ID_concessionaria = c.ID_concessionaria
JOIN AUTO a ON t.Targa = a.Targa
GROUP BY c.ID_concessionaria;

    YEAR(t.Data) AS Anno,
    SUM(CASE WHEN t.tipo_transazione = 'acquisto' THEN a.Prezzo ELSE 0 END) AS Fatturato_Acquisti,
    SUM(CASE WHEN t.tipo_transazione = 'vendita' THEN a.Prezzo ELSE 0 END) AS Fatturato_Vendite
FROM TRANSAZIONI t
JOIN CONCESSIONARIA c ON t.ID_concessionaria = c.ID_concessionaria
JOIN AUTO a ON t.Targa = a.Targa
GROUP BY c.ID_concessionaria, YEAR(t.Data);

SELECT *
FROM Fatturato
WHERE ID_concessionaria BETWEEN 1 AND 50
AND Anno BETWEEN 2010 AND 2015;

-- 13. Gestione Clienti per Concessionaria

CREATE VIEW Clienti AS
SELECT distinct
    c.ID_concessionaria,
    COUNT(CASE WHEN t.tipo_transazione = 'acquisto' THEN a.Prezzo ELSE 0 END) AS 'Clienti Acquisti',
    COUNT(CASE WHEN t.tipo_transazione = 'vendita' THEN a.Valore_vendita ELSE 0 END) AS 'Clienti Vendite'
FROM TRANSAZIONI t
JOIN CONCESSIONARIA c ON t.ID_concessionaria = c.ID_concessionaria
JOIN AUTO a ON t.Targa = a.Targa
GROUP BY c.ID_concessionaria;

SELECT *FROM Clienti
WHERE ID_concessionaria = 1;
